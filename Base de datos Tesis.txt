create database Tesis;

use Tesis


create table Carreras(Carrera varchar(45) primary key, Generacion varchar(20) UNIQUE);


create table Profesor(
NoCuentaProfesor int primary key, 
Nombre varchar(45),
Paterno varchar(45), 
Materno varchar(45), 
Sexo varchar(10), 
Celular bigint, 
CorreoElectronico varchar(60), 
CedulaProfesional varchar(50), 
Carrera varchar(45),
foreign key (Carrera)references Carreras(Carrera) on delete cascade on update cascade);


create table Alumnos(
NoCuentaAlumno int primary key, 
Nombre varchar(45),
Paterno varchar(45),
Materno varchar(45),
Sexo varchar(10),
Edad int, 
Direccion varchar(100), 
Celular bigint, 
CorreoElectronico varchar(60),
FechaNacimiento date, 
Carrera varchar(45), 
Generacion varchar(20),
foreign key (carrera)references carreras(carrera) on delete cascade on update cascade,
foreign key (Generacion)references carreras(Generacion) on delete cascade on update cascade);


create table Trabajos(IdTesis int primary key, 
NoCuentaAlumno int,
NoCuentaProfesor int,
Tema varchar(50),
Carrera varchar(45),
Generacion varchar(20),
FechaPresentacion date, 
FechaEntrega date, 
FechaPublicacion date,
foreign key (carrera)references carreras(carrera) on delete cascade on update cascade,
foreign key (Generacion)references carreras(Generacion) on delete cascade on update cascade);
